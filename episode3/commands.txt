# get unreserved server
curl -v -XGET localhost:8080/api/alice

# reserve server
curl -v -XPOST localhost:8080/api/alice

# release server
curl -v -XDELETE localhost:8080/api/alice

# (admin) get all servers
curl -v -XGET -H "X-ADMIN-TOKEN: SuperAdmin" localhost:8080/api/admin/servers

# (admin) release all servers
curl -v -XDELETE -H "X-ADMIN-TOKEN: SuperAdmin" localhost:8080/api/admin/servers
